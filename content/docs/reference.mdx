# Contract Reference

Reference documentation for Sew Protocol smart contracts.

## Core Contracts

### EscrowVault

The main escrow contract that holds funds and enforces release conditions.

```solidity
interface IEscrowVault {
    function createEscrow(
        address token,
        uint256 amount,
        address recipient,
        address releaseStrategy,
        address resolutionModule,
        uint256 timeout
    ) external returns (bytes32 escrowId);

    function release(bytes32 escrowId) external;

    function refund(bytes32 escrowId) external;

    function initiateDispute(bytes32 escrowId) external;

    function resolveDispute(bytes32 escrowId, uint256 recipientAmount) external;
}
```

### ModuleRegistry

Registry for approved release strategies and resolution modules.

```solidity
interface IModuleRegistry {
    function registerModule(address module, bytes32 moduleType) external;

    function isApprovedModule(address module, bytes32 moduleType) external view returns (bool);
}
```

## State Variables

```solidity
struct Escrow {
    address token;
    uint256 amount;
    address sender;
    address recipient;
    address releaseStrategy;
    address resolutionModule;
    EscrowState state;
    uint256 createdAt;
    uint256 timeout;
}

enum EscrowState {
    ACTIVE,
    RELEASED,
    REFUNDED,
    DISPUTED,
    RESOLVED
}
```
